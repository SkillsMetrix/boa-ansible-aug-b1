
- name: install lighttpd web server
  become: true
  apt:
    name: lighttpd
    state: latest
    update_cache: yes

- name: generate config file
  template:
    src: myconfig.j2
    dest: /etc/lighttpd/lighttpd.conf
  vars:
    server_port: 9090

- name: lighttpd start service
  service:
    name: lighttpd
    state: started
